# Files Tab + Word-like Editor Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a Files tab with a Word/Google-Docs-style rich-text editor that supports nested folders, autosaved rich documents, Ctrl+K AI palette (card insertion + block generation + analytic generation + heading insertion), grounded in the user's 77k-canonical card library and 39k-row analytics corpus, with docx export and native-Word-compatible clipboard.

**Architecture:** Vanilla JS client on top of Quill 2.x (CDN) with a small set of focused helper modules (`filesTree`, `filesEditor`, `filesOutline`, `filesAIPalette`). Express server adds a `docs` SQLite table and `/api/docs/*` routes; new `docsExport` and `docsAI` services reuse existing OpenRouter (`llm.js`), card retrieval (FTS + sqlite-vec), analytics FTS, and `clipboard.js` serializer. Server-side HTML→DOCX conversion uses the already-installed `docx` package.

**Tech Stack:** Node 18+, Express, better-sqlite3, sqlite-vec, `docx` npm package, OpenRouter via existing `server/services/llm.js` (DeepSeek V3.1 + Gemini 2.5 Flash-Lite), Quill 2.0 via CDN (`https://cdn.quilljs.com/2.0.0/quill.js` + `quill.snow.css`), vanilla JS client, `node --test` test runner.

```js
// server/services/docsStore.js
'use strict';
const crypto = require('crypto');
const { getDb } = require('./db');

function createDoc({ userId, kind, name, parentId = null, contentHtml = null }) {
  const id = crypto.randomUUID();
  const now = Date.now();
  getDb().prepare(`
    INSERT INTO docs (id, userId, parentId, kind, name, contentHtml, sortOrder, createdAt, updatedAt)
    VALUES (?, ?, ?, ?, ?, ?, 0, ?, ?)
  `).run(id, userId, parentId, kind, name, contentHtml, now, now);
  return getDoc(id, userId);
}

function getDoc(id, userId) {
  return getDb().prepare('SELECT * FROM docs WHERE id = ? AND userId = ?').get(id, userId) || null;
}

function listDocs(userId) {
  return getDb().prepare('SELECT id, parentId, kind, name, sortOrder, updatedAt FROM docs WHERE userId = ? ORDER BY sortOrder, name').all(userId);
}

function updateDoc(id, userId, patch) {
  const fields = [];
  const args = [];
  for (const k of ['name', 'parentId', 'contentHtml', 'sortOrder']) {
    if (k in patch) { fields.push(`${k} = ?`); args.push(patch[k]); }
  }
  if (!fields.length) return getDoc(id, userId);
  fields.push('updatedAt = ?'); args.push(Date.now());
  args.push(id, userId);
  getDb().prepare(`UPDATE docs SET ${fields.join(', ')} WHERE id = ? AND userId = ?`).run(...args);
  return getDoc(id, userId);
}

function deleteDoc(id, userId) {
  getDb().prepare('DELETE FROM docs WHERE id = ? AND userId = ?').run(id, userId);
}

module.exports = { createDoc, getDoc, listDocs, updateDoc, deleteDoc };
```

```js
// server/routes/docs.js
'use strict';
const express = require('express');
const store = require('../services/docsStore');

const router = express.Router();

function uid(req) { return req.user?.id || req.session?.user?.id; }

router.use((req, res, next) => {
  if (!uid(req)) return res.status(401).json({ error: 'auth_required' });
  next();
});

router.get('/', (req, res) => {
  res.json({ docs: store.listDocs(uid(req)) });
});

router.get('/:id', (req, res) => {
  const doc = store.getDoc(req.params.id, uid(req));
  if (!doc) return res.status(404).json({ error: 'not_found' });
  res.json({ doc });
});

router.post('/', (req, res) => {
  const { kind, name, parentId = null, contentHtml = null } = req.body || {};
  if (!kind || !name) return res.status(400).json({ error: 'kind_and_name_required' });
  if (kind !== 'folder' && kind !== 'file') return res.status(400).json({ error: 'bad_kind' });
  const doc = store.createDoc({ userId: uid(req), kind, name, parentId, contentHtml });
  res.json({ doc });
});

router.patch('/:id', (req, res) => {
  const existing = store.getDoc(req.params.id, uid(req));
  if (!existing) return res.status(404).json({ error: 'not_found' });
  const doc = store.updateDoc(req.params.id, uid(req), req.body || {});
  res.json({ doc });
});

router.delete('/:id', (req, res) => {
  store.deleteDoc(req.params.id, uid(req));
  res.json({ ok: true });
});

module.exports = router;
```

```js
// server/services/docsExport.js
'use strict';
const cheerio = require('cheerio');
const {
  Document, Packer, Paragraph, TextRun, HeadingLevel, BorderStyle,
} = require('docx');

function runsFromInline($, node) {
  const runs = [];
  const walk = (el, ctx) => {
    const $el = $(el);
    if (el.type === 'text') {
      if (!el.data) return;
      runs.push(new TextRun({
        text: el.data,
        bold: ctx.bold,
        italics: ctx.italic,
        underline: ctx.double ? { type: 'double' } : ctx.underline ? { type: 'single' } : undefined,
        highlight: ctx.highlight || undefined,
        font: ctx.font || 'Calibri',
        size: ctx.size ? ctx.size * 2 : undefined, // docx sizes are in half-points
      }));
      return;
    }
    const tag = (el.name || '').toLowerCase();
    const next = { ...ctx };
    if (tag === 'b' || tag === 'strong') next.bold = true;
    if (tag === 'i' || tag === 'em') next.italic = true;
    if (tag === 'u') next.underline = true;
    const style = $el.attr('style') || '';
    const bg = /background(?:-color)?:\s*(#[0-9a-f]{3,8}|cyan|yellow)/i.exec(style);
    if (bg) next.highlight = 'cyan';
    const fontMatch = /font-family:\s*([^;]+)/i.exec(style);
    if (fontMatch) next.font = fontMatch[1].replace(/['"]/g, '').split(',')[0].trim();
    const sizeMatch = /font-size:\s*(\d+)pt/i.exec(style);
    if (sizeMatch) next.size = Number(sizeMatch[1]);
    ($el.contents().toArray()).forEach(c => walk(c, next));
  };
  $(node).contents().toArray().forEach(c => walk(c, {}));
  return runs;
}

function paragraphFromBlock($, el, heading) {
  const runs = runsFromInline($, el);
  const opts = { children: runs };
  if (heading === 1) {
    opts.heading = HeadingLevel.HEADING_1;
    opts.border = {
      top:    { style: BorderStyle.SINGLE, size: 6, color: '000000' },
      bottom: { style: BorderStyle.SINGLE, size: 6, color: '000000' },
      left:   { style: BorderStyle.SINGLE, size: 6, color: '000000' },
      right:  { style: BorderStyle.SINGLE, size: 6, color: '000000' },
    };
  } else if (heading === 2) {
    opts.heading = HeadingLevel.HEADING_2;
    runs.forEach(r => { r.options.underline = { type: 'double' }; r.options.bold = true; });
  } else if (heading === 3) {
    opts.heading = HeadingLevel.HEADING_3;
    runs.forEach(r => { r.options.underline = { type: 'single' }; r.options.bold = true; });
  } else if (heading === 4) {
    opts.heading = HeadingLevel.HEADING_4;
    runs.forEach(r => { r.options.bold = true; });
  }
  return new Paragraph(opts);
}

async function htmlToDocxBuffer(html) {
  const $ = cheerio.load(`<body>${html}</body>`);
  const paragraphs = [];
  $('body').contents().each((_, el) => {
    if (el.type !== 'tag') return;
    const tag = el.name.toLowerCase();
    const h = { h1: 1, h2: 2, h3: 3, h4: 4 }[tag];
    if (h) { paragraphs.push(paragraphFromBlock($, el, h)); return; }
    if (tag === 'p' || tag === 'div') {
      paragraphs.push(paragraphFromBlock($, el, null));
      return;
    }
    // fallback: wrap unknown blocks
    paragraphs.push(paragraphFromBlock($, el, null));
  });
  const doc = new Document({ sections: [{ children: paragraphs }] });
  return Packer.toBuffer(doc);
}

module.exports = { htmlToDocxBuffer };
```

```js
// public/lib/filesTree.js
(function(global){
  'use strict';
  const API = global.API;

  function el(tag, attrs = {}, children = []) {
    const e = document.createElement(tag);
    Object.entries(attrs).forEach(([k,v]) => {
      if (k === 'class') e.className = v;
      else if (k.startsWith('on')) e.addEventListener(k.slice(2), v);
      else if (v != null) e.setAttribute(k, v);
    });
    (Array.isArray(children)?children:[children]).forEach(c => {
      if (c == null) return;
      if (typeof c === 'string') e.appendChild(document.createTextNode(c));
      else e.appendChild(c);
    });
    return e;
  }

  function byParent(rows) {
    const m = new Map();
    rows.forEach(r => {
      const k = r.parentId || '';
      if (!m.has(k)) m.set(k, []);
      m.get(k).push(r);
    });
    return m;
  }

  async function renderTree(container, onOpen) {
    const { docs } = await API.docsList();
    const bp = byParent(docs);
    container.innerHTML = '';
```

```js
const { getDb } = require('./db');

async function retrieveCards(query, k = 10) {
  // FTS match first
  const q = query.replace(/[^\w\s]/g,' ').trim();
  const rows = getDb().prepare(`
    SELECT c.id, c.tag, c.shortCite, substr(c.body_plain,1,400) AS body_plain,
           c.argumentTypes, c.argumentTags, c.typeLabel, c.topicLabel, bm25(cards_fts) AS rank
    FROM cards_fts JOIN cards c ON c.rowid = cards_fts.rowid
    WHERE cards_fts MATCH ? AND c.isCanonical = 1
    ORDER BY rank ASC
    LIMIT ?
  `).all(q, k).map(r => ({
    ...r,
    argumentTypes: safeJson(r.argumentTypes),
    argumentTags:  safeJson(r.argumentTags),
  }));
  return rows;
}

async function retrieveAnalytics(query, k = 5) {

    FROM analytics_fts JOIN analytics a ON a.rowid = analytics_fts.rowid
    WHERE analytics_fts MATCH ?
    ORDER BY rank ASC
    LIMIT ?
```

```js
    let idx = 0, offset = 0;

      if (offset + len > range.index) { idx = i; break; }
      offset += len;

      if (t === 'h2' && !result.h2) result.h2 = blocks[i].textContent;
      if (t === 'h3' && !result.h3) result.h3 = blocks[i].textContent;
      if (result.h1 && result.h2 && result.h3) break;

      if (kind === 'card') {
        const first = results.querySelector('.result');
        if (first) first.click();
        return;
      }
      if (kind === 'block' || kind === 'analytic') {
        results.innerHTML = '<div class="result">Generating…</div>';
        const headings = nearestHeadings(currentQuill);
        if (kind === 'block') {
          const out = await API.docsAiBlock({ intent: input.value, headings });
          insertBlock(out, sel.index);
        } else {
          const out = await API.docsAiAnalytic({ intent: input.value, headings });
          currentQuill.insertText(sel.index, '\n' + (out.text || '') + '\n');
        }
        close();
      }
    });
  }

  function renderCardResults(host, cards, insertAt) {
    host.innerHTML = '';
    cards.forEach(c => {
      const row = document.createElement('div');
      row.className = 'result';
      row.innerHTML = `<div style="font:700 12px/1.2 var(--font-sans)">${escapeHtml(c.tag||'')}</div>
        <div style="font:400 11px/1.2 var(--font-sans);color:#666">${escapeHtml(c.shortCite||'')}</div>`;
      row.addEventListener('click', () => {
        insertCard(c, insertAt);
        close();
      });
      host.appendChild(row);
    });
  }

  function insertCard(card, at) {
    const html = `<h4>${escapeHtml(card.tag||'')}</h4><p><b>${escapeHtml(card.shortCite||'')}</b></p><p>${escapeHtml(card.body_plain||'')}</p>`;
    currentQuill.insertEmbed(at, 'card-embed', { id: card.id, html }, 'user');
    currentQuill.setSelection(at + 1, 0);
  }

  function insertBlock(out, at) {
    let pos = at;
    if (out.analyticBefore) { currentQuill.insertText(pos, out.analyticBefore + '\n'); pos += out.analyticBefore.length + 1; }
    const picked = (out.candidateCards || []).filter(c => (out.pickedCardIds||[]).includes(c.id));
    const glue = out.glueBetween || [];
    picked.forEach((c,i) => {
      const html = `<h4>${escapeHtml(out.tag || c.tag || '')}</h4><p><b>${escapeHtml(c.shortCite||'')}</b></p><p>${escapeHtml(c.body_plain||'')}</p>`;
      currentQuill.insertEmbed(pos, 'card-embed', { id: c.id, html }, 'user');
      pos += 1;
      if (glue[i]) { currentQuill.insertText(pos, glue[i] + '\n'); pos += glue[i].length + 1; }
    });
    if (out.analyticAfter) { currentQuill.insertText(pos, out.analyticAfter + '\n'); pos += out.analyticAfter.length + 1; }
  }

  function escapeHtml(s){return String(s).replace(/[&<>"]/g,c=>({ '&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;'}[c]));}

  function close() { if (paletteEl) { paletteEl.remove(); paletteEl = null; } }

  function openMenu() {
    close();
    const sel = currentQuill.getSelection(true);
    paletteEl = document.createElement('div');
    paletteEl.className = 'files-palette';
    paletteEl.innerHTML = `
      <div class="results">
        <div class="result" data-k="card">Insert card</div>
        <div class="result" data-k="block">Generate block</div>
        <div class="result" data-k="analytic">Generate analytic</div>
      </div>`;
    document.body.appendChild(paletteEl);
    const bounds = currentQuill.getBounds(sel.index);
    const editorRect = currentQuill.root.getBoundingClientRect();
    paletteEl.style.left = Math.max(16, editorRect.left + bounds.left) + 'px';
    paletteEl.style.top  = (editorRect.top + bounds.top + 24 + window.scrollY) + 'px';
    paletteEl.querySelectorAll('.result').forEach(r => {
      r.addEventListener('click', () => open(r.dataset.k));
    });
    document.addEventListener('click', outsideClose, { once: true });
  }
  function outsideClose(e) { if (paletteEl && !paletteEl.contains(e.target)) close(); }

  function attach(quill) {
    currentQuill = quill;
    quill.keyboard.addBinding({ key: 'K', ctrlKey: true }, () => openMenu());
    quill.keyboard.addBinding({ key: 'K', metaKey: true }, () => openMenu());
  }

  global.FilesAIPalette = { attach };
})(window);
```
